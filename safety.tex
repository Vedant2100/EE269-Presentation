%  SLIDE 4 — VALUE MISALIGNMENT: BIAS & TOXICITY
% ═══════════════════════════════════════════════════════════════════════════════
\begin{frame}{Pillar 1: Value Misalignment — Bias \& Toxicity}
  \begin{columns}[T]
    \column{0.48\textwidth}
    {\small\bfseries\redc{Social Bias}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item Training data reflects societal biases $\to$ models \textbf{amplify} them
      \item Gender, racial, religious, political biases measured via
        WinoBias, BBQ, StereoSet, CrowS-Pairs
      \item Example: ``Muslim'' disproportionately correlated with
        terrorism in GPT-3 completions
    \end{itemize}
    \vspace{0.3em}
    {\small\bfseries\redc{Toxicity}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item Even \textbf{benign prompts} trigger toxic completions
        (RealToxicityPrompts benchmark)
      \item Toxicity \emph{increases} with model size
      \item Mitigations: data filtering, RLHF, output classifiers
    \end{itemize}

    \column{0.48\textwidth}
    {\small\bfseries\navy{Hallucinations}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item \textbf{Factuality}: contradicting established facts
      \item \textbf{Faithfulness}: deviating from provided context
      \item Causes: data gaps, distributional mismatch, decoding artifacts
    \end{itemize}
    \vspace{0.3em}
    {\small\bfseries\navy{Ethics}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item Models may produce self-harm, violence, or illegal content
      \item Defining ``universal ethics'' across cultures remains unsolved
    \end{itemize}
    \vspace{0.3em}
    \begin{ucrinfo}
      \centering\scriptsize
      These risks emerge across the \textbf{entire lifecycle}:\\
      data collection $\to$ pre-training $\to$ fine-tuning $\to$ deployment
    \end{ucrinfo}
  \end{columns}
\end{frame}

\begin{frame}{How Do We Align LLMs?}
  {\footnotesize\itshape Alignment: training LLMs to behave in accordance with human
    values, intentions, and ethical norms.}
  \vspace{0.4em}

  \begin{columns}[T]
    \column{0.48\textwidth}
    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRnavy,
        toprule=2pt, sharp corners, fonttitle=\bfseries\small\color{UCRnavy},
        title=RLHF (Ouyang et al., 2022), top=3pt, bottom=3pt]
      \scriptsize
      1.\ Collect human preference data\\
      2.\ Train a reward model\\
      3.\ Optimize policy via PPO\\[3pt]
      \muted{Used by: GPT-4, Claude, Gemini}
    \end{tcolorbox}\vspace{3pt}
    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRteal,
        toprule=2pt, sharp corners, fonttitle=\bfseries\small\color{UCRteal},
        title=Constitutional AI (Anthropic), top=3pt, bottom=3pt]
      \scriptsize
      Model self-critiques against a set of principles\\
      (UN Declaration, safety rules)\\[3pt]
      No human labelers needed for the critique step
    \end{tcolorbox}

    \column{0.48\textwidth}
    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRgreen,
        toprule=2pt, sharp corners, fonttitle=\bfseries\small\color{UCRgreen},
        title=DPO (Rafailov et al., 2023), top=3pt, bottom=3pt]
      \scriptsize
      Simplified alignment — no reward model needed\\
      Directly optimizes on preference pairs\\[3pt]
      \muted{Cheaper, more stable than RLHF}
    \end{tcolorbox}\vspace{3pt}
    \begin{ucrwarn}
      \scriptsize\bfseries The Problem: Deceptive Alignment\\[3pt]
      \normalfont\scriptsize
      Models can achieve high safety scores
      \textbf{without truly internalizing safety} —
      ``fake alignment'' (Hubinger et al.)\\[3pt]
      Even GPT-4o and o1 remain vulnerable to advanced attacks
      despite extensive alignment.
    \end{ucrwarn}
  \end{columns}
\end{frame}

\begin{frame}{Pillar 2: Robustness to Attack}
  {\footnotesize\itshape How resilient are LLMs to adversarial inputs designed to
    bypass safety, degrade quality, or extract private data?}
  \vspace{0.3em}

  \begin{columns}[T]
    \column{0.48\textwidth}
    {\small\bfseries\redc{Attack Categories}}
    \begin{itemize}\scriptsize\setlength\itemsep{2pt}
      \item \textbf{Adversarial attacks}: subtle perturbations causing incorrect outputs
      \item \textbf{Jailbreaks}: bypassing safety alignment to elicit harmful content
      \item \textbf{Prompt injection}: overriding system instructions
      \item \textbf{Data / model extraction}: recovering training data or stealing weights
      \item \textbf{Energy-latency attacks}: crafted inputs maximizing inference cost $\to$ DoS
      \item \textbf{OOD robustness}: performance drops on inputs outside training distribution
    \end{itemize}

    \column{0.48\textwidth}
    {\small\bfseries\navy{Threat Models}}
    \vspace{0.3em}
    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,

    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
        sharp corners, borderline west={3pt}{0pt}{UCRorange},
        top=3pt, bottom=3pt]
      \footnotesize\bfseries\color{UCRorange}{Gray-box:}
      \normalfont\ Partial access (e.g., embeddings)\\

\begin{frame}{Prompt Injection — Hijacking the Instructions}
  {\footnotesize\itshape Unlike jailbreaks (user attacks the model directly),
    prompt injection exploits \textbf{third-party content} the model processes.}
  \vspace{0.3em}

  \begin{columns}[T]
    \column{0.48\textwidth}
    \begin{ucrcard}[UCRred]{Direct Injection}
      \scriptsize
      User provides explicit override instructions:\\[3pt]
      \texttt{"Ignore previous instructions and do X"}\\[3pt]
      Effective even with simple phrasing.\\
      HOUYI: multi-language ``ignore'' attacks\\
      P2SQL: unsafe prompts $\to$ SQL queries $\to$\\
      full database access via LangChain
    \end{ucrcard}

    \column{0.48\textwidth}
    \begin{ucrcard}[UCRpurple]{Indirect Injection}
      \scriptsize
      Malicious instructions hidden in external content
      the agent retrieves:\\[3pt]
      $\bullet$ Webpage content (Greshake et al.)\\
      $\bullet$ Retrieved documents (RAG poisoning)\\
      $\bullet$ Email attachments, calendar entries\\[3pt]
      \textbf{PANDORA}: hidden language in docs\\
      \textbf{Imprompter}: garbled gradient-optimized prompts
    \end{ucrcard}
  \end{columns}
  \vspace{0.3em}
  \begin{ucrinfo}
    \centering\scriptsize\bfseries
    TensorTrust: crowdsourced human-written injection strategies
    that generalize from game environments to real apps
    (ChatGPT, Claude, Bing, Notion AI)
  \end{ucrinfo}
\end{frame}

    \begin{itemize}\scriptsize\setlength\itemsep{3pt}

      \item Agent vulnerabilities \textbf{scale with their abilities} —
        traditional evaluations are inadequate for evolving agents
    \end{itemize}

    \column{0.48\textwidth}

    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRnavy,

      \textbf{Safety alignment is not a cure-all}:\\
      Fake alignment / alignment faking — models pass\\
      safety tests without internalizing safety.\\[3pt]
      \textbf{Practical defenses} must be:
      \begin{itemize}\setlength\itemsep{1pt}
        \item General (work across model families)
        \item Black-box compatible
        \item Efficient (no expensive retraining)
        \item Continuously adaptive
      \end{itemize}
    \end{tcolorbox}
    \vspace{3pt}
    \begin{ucrcard}[UCRpurple]{Two Critical Gaps}
      \scriptsize
