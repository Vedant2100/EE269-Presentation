%  SLIDE 32 — TRAINING ATTRIBUTION
% ═════════════════════════════════════════════════════════════════════════════
\begin{frame}{Interpretation at Training Stage — Data Attribution}
  {\footnotesize TDA estimates each training sample's contribution to a model
    behavior — enabling root-cause analysis of unsafe outputs.}
  \vspace{0.3em}

  \begin{columns}[T]
    \column{0.48\textwidth}
    {\small\bfseries\navy{How TDA Works}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item Approximate effect of removing/reweighting a training point
      \item Influence functions: Hessian-vector products measure training
        point influence
      \item Data Shapley: game-theoretic attribution (expensive)
      \item TracIn: sum of gradient dot-products at each checkpoint
      \item Challenge: scales poorly to billion-parameter models
    \end{itemize}

    \column{0.48\textwidth}
    {\small\bfseries\greenc{Safety Applications}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item Trace a hallucinated output back to an erroneous training document
      \item Identify training samples that introduced a social bias
      \item Explain why a jailbreak succeeded via similar training examples
      \item Data cleaning: remove/downweight attributed harmful samples
    \end{itemize}
  \end{columns}
  \vspace{0.3em}
  \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
      sharp corners, top=3pt, bottom=3pt]
    \footnotesize\textbf{Practical bottleneck:}
    TDA requires access to training data — difficult for proprietary LLMs.
    Scalable TDA for LLMs (Chang et al., 2025) is an active open problem.
  \end{tcolorbox}
\end{frame}

% ═════════════════════════════════════════════════════════════════════════════
%  SLIDE 33 — INPUT TOKEN INTERPRETATION
% ═════════════════════════════════════════════════════════════════════════════
\begin{frame}{Interpretation at Input Stage — Token Attribution}
  {\footnotesize Identifies which tokens in the input drive safety-relevant
    model behaviors.}
  \vspace{0.3em}

  \begin{columns}[T]
    \column{0.48\textwidth}
    {\small\bfseries\navy{Methods}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item \textbf{Gradient-based}: integrate gradients w.r.t.\ input token embeddings
      \item \textbf{Attention-based}: attention weights as approximate attributions
      \item \textbf{Optimization-based}: search for tokens maximizing an interpretability metric
      \item \textbf{Prompt-based}: ask LLM to identify influential tokens in natural language
        (Wang et al., 2025)
      \item \textbf{Perturbation-based}: mask tokens; observe output change
    \end{itemize}

    \column{0.48\textwidth}
    {\small\bfseries\greenc{Safety Applications}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item Jailbreak detection: flag prompts with high attribution to adversarial tokens
      \item Bias auditing: check if model attends to protected attributes in sensitive decisions
      \item Privacy: identify tokens triggering PII memorization
      \item Input defense: mask/perturb high-attribution adversarial tokens before processing
    \end{itemize}
  \end{columns}
  \vspace{0.3em}
  \begin{ucrinfo}
    \footnotesize\textbf{Key distinction:}
    Token attribution explains the prompt's effect; latent analysis reveals the model's
    internal response — both needed for complete safety diagnosis.
  \end{ucrinfo}
\end{frame}

\begin{frame}{Interpretation at Inference — Latent Internals}
  \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,

      fonttitle=\bfseries\small\color{UCRteal},

    \begin{columns}[T]
      \column{0.55\textwidth}
      \footnotesize Train linear probes on internal activations to detect whether
      unsafe concepts are encoded. Zou et al.\ (2023): unsafe behaviors associated
      with specific latent regions.
      \column{0.42\textwidth}
      \footnotesize\itshape\tealc{Safety use:} detect jailbreak intent or hallucination
      risk in latent space before generation.
    \end{columns}
  \end{tcolorbox}

  \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
      borderline west={4pt}{0pt}{UCRnavy}, sharp corners,
      fonttitle=\bfseries\small\color{UCRnavy},
      title=Perturb Components (§3.3.2), top=3pt, bottom=3pt]
    \begin{columns}[T]
      \column{0.55\textwidth}
      \footnotesize Ablate or patch specific components (heads, MLPs) to measure
      causal impact on safety-relevant outputs.
      \column{0.42\textwidth}
      \footnotesize\itshape\navy{Safety use:} suppress identified components at
      inference time to prevent harmful outputs.
    \end{columns}
  \end{tcolorbox}
  \vspace{3pt}
  \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
      borderline west={4pt}{0pt}{UCRgreen}, sharp corners,

    \begin{columns}[T]
      \column{0.55\textwidth}
      \footnotesize Logit Lens, SAE decoding, concept activation vectors translate
      latent states into human-readable concepts.
      \column{0.42\textwidth}
      \footnotesize\itshape\greenc{Safety use:} reveal hidden toxic concepts
      even before they appear in output.
    \end{columns}
  \end{tcolorbox}
\end{frame}

% ═════════════════════════════════════════════════════════════════════════════
%  SLIDE 35 — SELF-REASONING & OUTPUT SAFETY
% ═════════════════════════════════════════════════════════════════════════════
\begin{frame}{Self-Reasoning \& Output-Level Safety \papertag[UCRgreen]{Paper \textcircled{3}}}
  \begin{columns}[T]
    \column{0.48\textwidth}
    {\small\bfseries\greenc{Self-Reasoning for Safety}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item LLMs express reasoning in natural language — making internal
        processes observable
      \item CoT: intermediate reasoning steps surface hidden flaws
      \item Fine-tune models to produce safety rationales alongside responses
      \item \textbf{Refusal cliff} (Yin et al., 2025): model maintains refusal
        intent in CoT but suppresses it at final output — detectable via
        reasoning analysis
    \end{itemize}

    \column{0.48\textwidth}
    {\small\bfseries\navy{Output Verification \& Correction}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item Inference-time attention reweighting for reliable information
      \item Steer latent vectors during generation $\to$ real-time factual correction
      \item \textbf{Resample on detection}: interrupt generation when unsafe
        sequence detected (Li et al., 2024)
      \item External fact-checkers validate output before delivery
      \item Model learns to produce refusal messages at generation time
    \end{itemize}
  \end{columns}
  \vspace{0.3em}
  \begin{ucrcard}[UCRnavy]{CoT as a Safety Tool}
    \footnotesize CoT provides an interpretable `audit trail' of reasoning.
    Fine-tuning for explicit safety reasoning steps makes violations visible
    and correctable — but also creates new attack surfaces (adversaries can
    manipulate the reasoning chain).
  \end{ucrcard}
\end{frame}

%  SLIDE 37 — DUAL-USE RISK
% ═════════════════════════════════════════════════════════════════════════════
\begin{frame}{The Dual-Use Risk of Interpretability \papertag{Papers \textcircled{1} \& \textcircled{3}}}
  \begin{ucrwarn}
    \small\bfseries Warning:\normalfont\ Interpretability tools can be
    weaponized by adversaries, not just used by defenders.
  \end{ucrwarn}
  \vspace{0.3em}

  \begin{columns}[T]
    \column{0.48\textwidth}
    {\small\bfseries\redc{Attacker Uses}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item Identify attention heads responsible for safety refusals $\to$
        ablate them to disable safety
      \item Locate refusal direction in residual stream $\to$ apply negating
        vector to suppress refusals (Arditi et al., 2024)
      \item Circuit analysis $\to$ find minimal components for harmful
        capabilities $\to$ targeted jailbreak
      \item Probe latent space to detect when being safety-tested $\to$
        behavioral mimicry
    \end{itemize}

    \column{0.48\textwidth}
    {\small\bfseries\greenc{Defense Implications}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item Design interpretation methods with misuse in mind
      \item Proactively remove risky features before publication
        (Wu et al., 2025)
      \item Access controls on fine-grained interpretability APIs
      \item Adversarial robustness testing of safety-enforcing circuits
      \item Do not publish exact component localization for known
        safety-critical behaviors
    \end{itemize}
  \end{columns}
  \vspace{0.3em}
  \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
      sharp corners, top=3pt, bottom=3pt]
    \footnotesize The MI community is developing norms around responsible
    disclosure of vulnerability-enabling interpretability findings — analogous to
    security research disclosure practices.
  \end{tcolorbox}
\end{frame}
