\begin{frame}{Why LLM Safety Is Urgent}
  \vspace*{0.1cm}
  {\small\itshape LLMs are deployed in healthcare, law, finance, education —
    errors can have real-world consequences.}
  \vspace{0.4em}

  \begin{columns}[T]
    \column{0.32\textwidth}
    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
        borderline west={3pt}{0pt}{UCRnavy}, sharp corners,
        fonttitle=\bfseries\small\color{UCRnavy}, title=Bias \& Discrimination,
        height=3.3cm, valign=top, top=4pt, bottom=4pt, left=5pt]
      \footnotesize Models replicate and amplify social stereotypes;
      performance disparities across ethnic/language groups cause unjust outcomes
      in high-stakes domains.
    \end{tcolorbox}

    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
        borderline west={3pt}{0pt}{UCRteal}, sharp corners,
        fonttitle=\bfseries\small\color{UCRteal}, title=Adversarial Attacks,
        height=3.3cm, valign=top, top=4pt, bottom=4pt, left=5pt]
      \footnotesize Carefully crafted prompts (jailbreaks) bypass safety
      alignment; white-box gradient attacks can force harmful outputs from
      aligned models.
    \end{tcolorbox}

    \column{0.32\textwidth}
    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
        borderline west={3pt}{0pt}{UCRred}, sharp corners,
        fonttitle=\bfseries\small\color{UCRred}, title=Privacy Leakage,
        height=3.3cm, valign=top, top=4pt, bottom=4pt, left=5pt]
      \footnotesize Training data memorization enables extraction of
      personal information; LLMs can be prompted to expose sensitive user data.
    \end{tcolorbox}

    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
        borderline west={3pt}{0pt}{UCRpurple}, sharp corners,
        fonttitle=\bfseries\small\color{UCRpurple}, title=Misuse at Scale,
        height=3.3cm, valign=top, top=4pt, bottom=4pt, left=5pt]
      \footnotesize Automated misinformation, deepfakes, and weapons-design
      assistance — at a scale no prior technology enabled.
    \end{tcolorbox}

    \column{0.32\textwidth}
    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
        borderline west={3pt}{0pt}{UCRforest}, sharp corners,
        fonttitle=\bfseries\small\color{UCRforest}, title=Autonomous AI Risk,
        height=3.3cm, valign=top, top=4pt, bottom=4pt, left=5pt]
      \footnotesize Increasingly agentic LLMs may exhibit goal misalignment,
      deception, and situational awareness in unexpected and hard-to-detect ways.
    \end{tcolorbox}

    \begin{tcolorbox}[enhanced, colback=UCRpalered, colframe=UCRred,
        sharp corners,
        fonttitle=\bfseries\small\color{UCRred}, title=No Simple Fix,
        height=3.3cm, valign=top, top=4pt, bottom=4pt, left=5pt]
      \footnotesize Safety issues \textbf{cannot} be resolved by scaling models
      alone \mbox{(Wei et al., 2023)}. New principled methods are needed.
    \end{tcolorbox}
  \end{columns}
\end{frame}
