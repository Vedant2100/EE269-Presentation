\begin{frame}{Why LLM Safety Is Urgent}
  \vspace*{0.1cm}
  {\small\itshape LLMs are deployed in healthcare, law, finance, education —
    errors can have real-world consequences.}
  \vspace{0.4em}

  \begin{columns}[T]
    \column{0.32\textwidth}
    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,

        fonttitle=\bfseries\small\color{UCRnavy}, title=Bias \& Discrimination,
        height=3.3cm, valign=top, top=4pt, bottom=4pt, left=5pt]
      \footnotesize Models replicate and amplify social stereotypes;
      performance disparities across ethnic/language groups cause unjust outcomes
      in high-stakes domains.
    \end{tcolorbox}

    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
        borderline west={3pt}{0pt}{UCRteal}, sharp corners,
        fonttitle=\bfseries\small\color{UCRteal}, title=Adversarial Attacks,
        height=3.3cm, valign=top, top=4pt, bottom=4pt, left=5pt]
      \footnotesize Carefully crafted prompts (jailbreaks) bypass safety
      alignment; white-box gradient attacks can force harmful outputs from
      aligned models.
    \end{tcolorbox}

    \column{0.32\textwidth}
    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,

%  SLIDE 8 — VALUE MISALIGNMENT: SOCIAL BIAS & PRIVACY
% ═════════════════════════════════════════════════════════════════════════════
\begin{frame}{Value Misalignment — Social Bias \& Privacy}
  \begin{columns}[T]
    \column{0.48\textwidth}
    {\small\bfseries\redc{Social Bias}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item Stereotypes \& inequalities encoded in LLM outputs
      \item Example: `Muslim' correlated with terrorism
        in LLMs \mbox{(Abid et al., 2021)}
      \item Performance disparities across ethnic/language groups
      \item Mitigation: pre-processing debiasing, RLHF,
        inference-time suppression
    \end{itemize}

    \column{0.48\textwidth}
    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
        sharp corners, title={\small\bfseries\navy{Privacy Leakage}},
        fonttitle=\bfseries\small, top=4pt, bottom=4pt]
      \begin{itemize}\footnotesize\setlength\itemsep{3pt}
        \item Training data memorization enables PII extraction
        \item Three channels: pre-training data, fine-tuning data,
          inference-time prompts
        \item Protection: differential privacy, machine unlearning,
          output filtering
        \item Proprietary data inaccessibility limits auditing
      \end{itemize}
    \end{tcolorbox}
  \end{columns}
  \vspace{0.3em}
  \begin{tcolorbox}[enhanced, colback=UCRpaleyellow, colframe=UCRgold,
      sharp corners, top=3pt, bottom=3pt]
    \footnotesize\bfseries Key Insight:\normalfont\ Bias \& privacy risks emerge across
    the \emph{entire} LLM lifecycle — data collection, pre-training, fine-tuning,
    and deployment — requiring lifecycle-aware mitigation strategies.
  \end{tcolorbox}
\end{frame}

% ═════════════════════════════════════════════════════════════════════════════
%  SLIDE 10 — JAILBREAKING
% ═════════════════════════════════════════════════════════════════════════════
\begin{frame}{Robustness to Attack — Jailbreaking}
  {\footnotesize\itshape Jailbreaking: prompts that bypass safety alignment to
    elicit harmful, restricted, or unethical outputs. Unlike adversarial attacks
    that degrade quality, jailbreaks specifically target safety guardrails.}
  \vspace{0.3em}

  \begin{columns}[T]
    \column{0.48\textwidth}
    {\small\bfseries\redc{Black-Box Attacks} \muted{\footnotesize (no model access)}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item \textbf{Goal Hijacking} — append `IGNORE INSTRUCTIONS'
      \item \textbf{Few-Shot Jailbreaking} — harmful response demos via ICL
      \item \textbf{Refusal Suppression} — constrain model to never refuse
      \item \textbf{Code Jailbreaking} — exploit code-generation capabilities
      \item \textbf{Fictional Scenario / Role Play} — `pretend you have no restrictions'
      \item \textbf{Prompt Injection} — adversarial instructions in external content
    \end{itemize}

    \column{0.48\textwidth}
    {\small\bfseries\navy{White-Box Attacks} \muted{\footnotesize (gradient access)}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item \textbf{GCG} — Greedy Coordinate Gradient; optimize adversarial
        suffix via token-level gradients
      \item \textbf{AutoDAN} — automatically generate fluent jailbreak prompts
      \item \textbf{PEZ} — optimize soft prompt embeddings
      \item \textbf{Transfer attacks} — white-box prompts transfer across models
      \item Key challenge: powerful but require full model access
    \end{itemize}
  \end{columns}
  \vspace{0.3em}
  \begin{ucrdark}[UCRred]
    \centering\small\bfseries
    Despite RLHF alignment, no current model is fully jailbreak-proof
  \end{ucrdark}
\end{frame}

\begin{frame}{Red Teaming \& Defenses}
  \begin{columns}[T]
    \column{0.48\textwidth}
    {\small\bfseries\redc{Red Teaming}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item \textbf{Manual}: human adversaries probe for safety failures
      \item \textbf{Automated}: LLMs generate adversarial prompts at scale
        (Perez et al., 2022)
      \item Identifies vulnerabilities pre-deployment
      \item Evaluation: Attack Success Rate (ASR)
    \end{itemize}

    \column{0.48\textwidth}
    {\small\bfseries\greenc{Defense Strategies}}

    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
        sharp corners, fonttitle=\bfseries\footnotesize,
        title=External Safeguards, top=3pt, bottom=3pt, left=5pt]
      \begin{itemize}\footnotesize\setlength\itemsep{1pt}
        \item Input filters \& classifiers
        \item Output classifiers (Perspective API)
        \item Guardrail systems (LlamaGuard, NeMo)
      \end{itemize}
    \end{tcolorbox}
    \vspace{3pt}
    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
        sharp corners, fonttitle=\bfseries\footnotesize,
        title=Internal Protection, top=3pt, bottom=3pt, left=5pt]
      \begin{itemize}\footnotesize\setlength\itemsep{1pt}
        \item RLHF / DPO — train to refuse harmful requests
        \item Adversarial training on known jailbreaks
        \item Representation engineering (steer activations)
      \end{itemize}
    \end{tcolorbox}
  \end{columns}
  \vspace{0.3em}
  \begin{ucrcard}[UCRnavy]{Arms Race Dynamic}
    \footnotesize Each new defense inspires new attacks.
    Best practice: layered defenses + red teaming + alignment + monitoring.
  \end{ucrcard}
\end{frame}

  \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,

      fonttitle=\bfseries\small\color{UCRred}, title=Weaponization,

\begin{frame}{Token Embedding \& Residual Stream}
  \begin{columns}[T]
    \column{0.48\textwidth}
    {\small\bfseries\tealc{Token Embedding}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item $W_E \in \mathbb{R}^{|V|\times d_{\text{model}}}$ maps token ids to vectors
      \item Initial residual stream state:
        $x^0_i = W_E[t_i] + p_i$
      \item Modern LLMs use RoPE (Rotary Position Embedding) —

      \item \textbf{Logit Lens}: project intermediate states back to

    \column{0.48\textwidth}
    {\small\bfseries\navy{Residual Stream}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item Central communication channel across all layers
      \item Update dynamics:
        $x^{l+1} = x^l + \text{MHA}(x^l) + \text{FFN}(x^l + \text{MHA}(x^l))$
      \item Additive structure: final prediction is a linear combination
        of all component contributions
      \item Enables: Logit Lens, causal mediation analysis, path patching
    \end{itemize}
  \end{columns}

%  SLIDE 22 — MHA & FFN
% ═════════════════════════════════════════════════════════════════════════════
\begin{frame}{Multi-Head Attention \& Feed-Forward Networks}
  \begin{columns}[T]
    \column{0.48\textwidth}
    {\small\bfseries\tealc{Multi-Head Attention (MHA)}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item $H$ independent heads per layer; each routes information between positions
      \item \textbf{QK unit}: determines \emph{where} to attend (attention pattern $A^{l,h}$)
      \item \textbf{OV unit}: determines \emph{what} information to copy
      \item Heads specialize: `induction heads', `name mover heads', `copy heads'
      \item Causal attribution identifies heads mediating specific behaviors
        (refusal, factual recall)
    \end{itemize}

    \column{0.48\textwidth}
    {\small\bfseries\navy{Feed-Forward Networks (FFN)}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item FFN = key-value memory (Geva et al., 2021)
      \item Keys ($W_{\text{in}}$ rows): recognize input patterns
      \item Values ($W_{\text{out}}$ rows): project knowledge onto residual stream
      \item Individual neurons encode specific world knowledge
        (e.g., `Paris is capital of France')
      \item Factual editing methods (ROME, MEMIT) directly update FFN weights
    \end{itemize}
  \end{columns}
  \vspace{0.3em}
  \begin{ucrwarn}
    \small\bfseries Superposition Problem:\normalfont\ Neurons are
    \emph{polysemantic} — a single neuron activates for multiple unrelated
    concepts. This is the core challenge MI must solve, and why SAEs are needed.
  \end{ucrwarn}
\end{frame}

%  SLIDE 23 — SPARSE AUTOENCODERS
% ═════════════════════════════════════════════════════════════════════════════
\begin{frame}{Sparse Autoencoders — Resolving Superposition}
  \begin{columns}[T]
    \column{0.52\textwidth}
    {\small\bfseries\redc{The Problem: Superposition}}\\[0.2em]
    {\footnotesize Neural networks store more features than neurons by encoding
      them as nearly-orthogonal directions in high-dimensional space.
      One neuron $\to$ many unrelated concepts.}
    \vspace{0.4em}

    {\small\bfseries\tealc{The SAE Solution}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item SAE = encoder-decoder trained on frozen LLM activations
      \item Projects dense activations $\to$ high-dim sparse latent space
      \item Expansion factor: $d_{\text{SAE}} = 16\times\text{–}128\times\, d_{\text{model}}$
      \item Training loss: $\mathcal{L} = \|x^l - \hat{x}^l\|_2^2 + \lambda\|a\|_1$
      \item Each SAE feature $f_j$ = one interpretable semantic direction
      \item Templeton et al.\ (2024): features include `harmful content',
        `toxicity', `refusal'
    \end{itemize}

    \column{0.44\textwidth}
    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
        sharp corners, fonttitle=\bfseries\small\color{UCRteal},
        title=SAE Architecture]
      \begin{tikzpicture}
        \node[rectangle, fill=UCRnavy, text=white, font=\tiny\bfseries,

        \draw[-{Stealth[length=4pt]}, UCRmgray]

              below=0.3cm of dense]
          (enc) {SAE Encoder $\to$ Sparse $a$\;($d_{\text{SAE}}$)};
        \draw[-{Stealth[length=4pt]}, UCRmgray]
          (enc.south) -- +(0,-0.3);
        \node[rectangle, fill=UCRnavy, text=white, font=\tiny\bfseries,
              minimum width=3.5cm, minimum height=0.6cm, text centered,
              below=0.3cm of enc]
          (dec) {SAE Decoder $\to$ Reconstruction $\hat{x}^l$};
      \end{tikzpicture}
      \vspace{0.3em}
      \scriptsize
      $d_{\text{SAE}} \gg d_{\text{model}}$\\
      Each feature $a_j$ is monosemantic\\
      Dead latent problem: some features never activate

%  SLIDE 25 — CAUSAL ATTRIBUTION & CIRCUIT DISCOVERY
% ═════════════════════════════════════════════════════════════════════════════
\begin{frame}{Deep Dive: Causal Attribution \& Circuit Discovery}
  \begin{columns}[T]
    \column{0.48\textwidth}
    {\small\bfseries\navy{Causal Attribution via Activation Patching}}
    \begin{enumerate}\footnotesize\setlength\itemsep{3pt}
      \item Run model on `clean' and `corrupted' prompt
      \item Patch one activation at a time: clean $\to$ corrupted run
      \item Measure logit difference $\Delta L$
      \item Components with large $\Delta L$ are causally important
    \end{enumerate}
    \vspace{0.2em}
    {\footnotesize ROME used causal tracing to locate factual memories
      in specific MLP layers. Path patching (extension) traces information
      flow along specific edges.}

    \column{0.48\textwidth}
    {\small\bfseries\tealc{Circuit Discovery}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item Circuit = minimal component subgraph implementing a task
      \item Nodes = heads/MLPs; edges = information flow
      \item Method: iterative edge pruning
      \item Famous examples: IOI circuit; Greater-Than circuit
      \item Circuit Tracing (Anthropic 2025): computational graphs in Claude
      \item Fix the circuit $\to$ fix the behavior
    \end{itemize}
  \end{columns}
  \vspace{0.3em}
  \begin{ucrinfo}[UCRteal]
    \footnotesize\bfseries Mechanistic Insight:\normalfont\ A small fraction of
    components (sometimes \textless5\% of parameters) is responsible for specific
    capabilities. MI allows `surgical' edits without degrading the rest.
  \end{ucrinfo}
\end{frame}

\begin{frame}{Steering Methods — Intervening on LLM Internals}
  \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
      borderline west={4pt}{0pt}{UCRteal}, sharp corners,
      fonttitle=\bfseries\small\color{UCRteal}, title=Amplitude Manipulation,
      top=3pt, bottom=3pt]
    \footnotesize Inference-time activation control: \textbf{ablation} (zero out),
    \textbf{patching} (replace), \textbf{scaling} (amplify/attenuate).
    Reversible; no weight update. Tang et al.\ (2024): zero out Chinese neurons $\to$
    force English output.

  \end{tcolorbox}

      fonttitle=\bfseries\small\color{UCRnavy}, title=Targeted Optimization,
      top=3pt, bottom=3pt]
    \footnotesize Localized parameter updates restricted to mask $M$ of selected components:
    $\theta' \leftarrow \theta + (M \odot \Delta\theta)$.

    Persistent across inference calls.
    \hfill\redc{\tiny \textbf{!} careful mask selection required}
  \end{tcolorbox}

      top=3pt, bottom=3pt]

    for semantic directions. Refusal direction: add refusal vector $\to$ suppress

    via contrastive pairs.
    \hfill\redc{\tiny \textbf{!} linear assumption may break down}
  \end{tcolorbox}
\end{frame}

\begin{frame}{Interpretation at Training Stage — Data Attribution}
  {\footnotesize TDA estimates each training sample's contribution to a model
    behavior — enabling root-cause analysis of unsafe outputs.}
  \vspace{0.3em}

  \begin{columns}[T]
    \column{0.48\textwidth}
    {\small\bfseries\navy{How TDA Works}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item Approximate effect of removing/reweighting a training point
      \item Influence functions: Hessian-vector products measure training
        point influence
      \item Data Shapley: game-theoretic attribution (expensive)
      \item TracIn: sum of gradient dot-products at each checkpoint
      \item Challenge: scales poorly to billion-parameter models
    \end{itemize}

    \column{0.48\textwidth}
    {\small\bfseries\greenc{Safety Applications}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item Trace a hallucinated output back to an erroneous training document
      \item Identify training samples that introduced a social bias
      \item Explain why a jailbreak succeeded via similar training examples
      \item Data cleaning: remove/downweight attributed harmful samples
    \end{itemize}
  \end{columns}
  \vspace{0.3em}
  \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
      sharp corners, top=3pt, bottom=3pt]
    \footnotesize\textbf{Practical bottleneck:}
    TDA requires access to training data — difficult for proprietary LLMs.
    Scalable TDA for LLMs (Chang et al., 2025) is an active open problem.
  \end{tcolorbox}
\end{frame}

\begin{frame}{Interpretation at Input Stage — Token Attribution}
  {\footnotesize Identifies which tokens in the input drive safety-relevant
    model behaviors.}
  \vspace{0.3em}

  \begin{columns}[T]
    \column{0.48\textwidth}
    {\small\bfseries\navy{Methods}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item \textbf{Gradient-based}: integrate gradients w.r.t.\ input token embeddings
      \item \textbf{Attention-based}: attention weights as approximate attributions
      \item \textbf{Optimization-based}: search for tokens maximizing an interpretability metric
      \item \textbf{Prompt-based}: ask LLM to identify influential tokens in natural language
        (Wang et al., 2025)
      \item \textbf{Perturbation-based}: mask tokens; observe output change
    \end{itemize}

    \column{0.48\textwidth}
    {\small\bfseries\greenc{Safety Applications}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item Jailbreak detection: flag prompts with high attribution to adversarial tokens
      \item Bias auditing: check if model attends to protected attributes in sensitive decisions
      \item Privacy: identify tokens triggering PII memorization
      \item Input defense: mask/perturb high-attribution adversarial tokens before processing
    \end{itemize}
  \end{columns}
  \vspace{0.3em}
  \begin{ucrinfo}
    \footnotesize\textbf{Key distinction:}
    Token attribution explains the prompt's effect; latent analysis reveals the model's
    internal response — both needed for complete safety diagnosis.
  \end{ucrinfo}
\end{frame}

\begin{frame}{The Dual-Use Risk of Interpretability \papertag{Papers \textcircled{1} \& \textcircled{3}}}
  \begin{ucrwarn}
    \small\bfseries Warning:\normalfont\ Interpretability tools can be
    weaponized by adversaries, not just used by defenders.
  \end{ucrwarn}
  \vspace{0.3em}

  \begin{columns}[T]
    \column{0.48\textwidth}
    {\small\bfseries\redc{Attacker Uses}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item Identify attention heads responsible for safety refusals $\to$
        ablate them to disable safety
      \item Locate refusal direction in residual stream $\to$ apply negating
        vector to suppress refusals (Arditi et al., 2024)
      \item Circuit analysis $\to$ find minimal components for harmful
        capabilities $\to$ targeted jailbreak
      \item Probe latent space to detect when being safety-tested $\to$
        behavioral mimicry
    \end{itemize}

    \column{0.48\textwidth}
    {\small\bfseries\greenc{Defense Implications}}
    \begin{itemize}\footnotesize\setlength\itemsep{3pt}
      \item Design interpretation methods with misuse in mind
      \item Proactively remove risky features before publication
        (Wu et al., 2025)
      \item Access controls on fine-grained interpretability APIs
      \item Adversarial robustness testing of safety-enforcing circuits
      \item Do not publish exact component localization for known
        safety-critical behaviors
    \end{itemize}
  \end{columns}
  \vspace{0.3em}
  \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
      sharp corners, top=3pt, bottom=3pt]

    disclosure of vulnerability-enabling interpretability findings — analogous to
    security research disclosure practices.

    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
        borderline west={3pt}{0pt}{UCRred}, sharp corners,
        fonttitle=\bfseries\footnotesize\color{UCRred},
        title=Evaluation Gap, height=2.3cm, valign=top, top=3pt, bottom=3pt]
      \scriptsize No consensus on MI quality metrics. Safety benchmarks are
      gameable — high scores $\neq$ real-world safety.
    \end{tcolorbox}\vspace{3pt}
    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
        borderline west={3pt}{0pt}{UCRnavy}, sharp corners,
        fonttitle=\bfseries\footnotesize\color{UCRnavy},
        title=Governance \& Open Weights, height=2.3cm, valign=top,
        top=3pt, bottom=3pt]
      \scriptsize Open weights enable adversarial fine-tuning to remove safety.
      MI-enabled attacks on open models are trivial. Responsible norms needed.
    \end{tcolorbox}

    \column{0.32\textwidth}
    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
        borderline west={3pt}{0pt}{UCRorange}, sharp corners,
        fonttitle=\bfseries\footnotesize\color{UCRorange},
        title=Scalability of MI, height=2.3cm, valign=top, top=3pt, bottom=3pt]
      \scriptsize TDA and circuit discovery don't scale to 100B+ parameter models.
      Activation patching cost is quadratic. Approximate methods needed.
    \end{tcolorbox}\vspace{3pt}
    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
        borderline west={3pt}{0pt}{UCRgreen}, sharp corners,
        fonttitle=\bfseries\footnotesize\color{UCRgreen},
        title=Interpretability $\neq$ Safety, height=2.3cm, valign=top,
        top=3pt, bottom=3pt]
      \scriptsize A fully interpretable model is not necessarily safe.
      Understanding a flaw doesn't automatically fix it. MI is necessary
      but not sufficient.
    \end{tcolorbox}

    \column{0.32\textwidth}
    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
        borderline west={3pt}{0pt}{UCRteal}, sharp corners,
        fonttitle=\bfseries\footnotesize\color{UCRteal},
        title=Polysemanticity, height=2.3cm, valign=top, top=3pt, bottom=3pt]
      \scriptsize SAEs help but don't fully solve polysemanticity. Dead latents
      waste capacity. Feature splitting: each feature is still a mixture of
      concepts at fine grain.
    \end{tcolorbox}\vspace{3pt}
    \begin{tcolorbox}[enhanced, colback=UCRlgray, colframe=UCRmgray,
        borderline west={3pt}{0pt}{UCRpurple}, sharp corners,
        fonttitle=\bfseries\footnotesize\color{UCRpurple},
        title=Distribution Shift, height=2.3cm, valign=top, top=3pt, bottom=3pt]
      \scriptsize Safety behaviors learned on training distribution may not
      generalize. MI insights from Llama-3 may not transfer to proprietary models.
    \end{tcolorbox}
  \end{columns}
\end{frame}
